Participación Electoral por Comuna

Vamos a calcular y visualizar la participación electoral por comuna. Para esto, necesitamos: 1. Calcular el total de electores por comuna 2. Calcular los votos emitidos por comuna 3. Calcular el porcentaje de participación

Code 
# Calculamos participación por comuna
participacion_comuna <- df_caba %>%
  group_by(seccion_nombre, mesa_id) %>%
  summarise(
    electores = first(mesa_electores), #eliminamos duplicaods en mesa_electores
    votos_emitidos = sum(votos_cantidad),
    .groups = "drop"
  ) %>%
  group_by(seccion_nombre) %>%
  summarise(
    total_electores = sum(electores),
    total_votos = sum(votos_emitidos),
    participacion = (total_votos / total_electores) * 100,
    .groups = "drop"
  ) %>%
  arrange(desc(participacion))

# Visualizamos la participación por comuna
ggplot(participacion_comuna, 
       aes(x = reorder(seccion_nombre, participacion), 
           y = participacion)) +
  geom_col(fill = "steelblue", alpha = 0.7) +
  geom_text(aes(label = sprintf("%.1f%%", participacion)), 
            vjust = -0.5) +
  labs(title = "Participación Electoral por Comuna",
       subtitle = "Elecciones CABA 2023 - Diputados/as",
       x = "Comuna",
       y = "Participación (%)") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
  scale_y_continuous(limits = c(0, max(participacion_comuna$participacion) * 1.1))

Composición del Voto

Analizaremos la composición del voto en toda la Ciudad, incluyendo votos positivos, en blanco, nulos y otros tipos de votos.

Code 
# Calculamos totales por tipo de voto
composicion_votos <- df_caba %>%
  group_by(votos_tipo) %>%
  summarise(
    total_votos = sum(votos_cantidad),
    .groups = "drop"
  ) %>%
  mutate(
    porcentaje = total_votos / sum(total_votos) * 100,
    votos_tipo = factor(votos_tipo, 
                       levels = c("POSITIVO", "EN BLANCO", "NULO", 
                                "IMPUGNADO", "RECURRIDO", "COMANDO"))

  arrange(desc(porcentaje))

# Creamos un gráfico de torta

  coord_polar("y", start = 0) +
  scale_fill_brewer(palette = "Set3") +
  geom_text(aes(label = sprintf("%.1f%%", porcentaje)),
            position = position_stack(vjust = 0.5)) +
  labs(title = "Composición del Voto en CABA",
       subtitle = "Elecciones 2023 - Diputados/as",

  theme_void() +

Resultados por Partido Político

Analizaremos los resultados obtenidos por cada agrupación en las diferentes comunas de la Ciudad.

Code 
# Calculamos votos por partido y comuna
resultados_partido <- df_caba %>%
  filter(
    votos_tipo == "POSITIVO",
    !is.na(agrupacion_nombre)
  ) %>%
  group_by(seccion_nombre, agrupacion_nombre) %>%
  summarise(
    votos = sum(votos_cantidad),
    .groups = "drop"
  ) %>%
  group_by(seccion_nombre) %>%
  mutate(
    porcentaje = votos / sum(votos) * 100,
    seccion_nombre = factor(seccion_nombre, 
                          levels = unique(seccion_nombre)[order(unique(seccion_nombre))])
  ) %>%
  ungroup()

# Crear gráfico de barras apiladas
ggplot(resultados_partido, 
       aes(x = seccion_nombre, y = porcentaje, fill = agrupacion_nombre)) +
  geom_col() +
  geom_text(aes(label = sprintf("%.1f%%", porcentaje)),
            position = position_stack(vjust = 0.5),
            size = 3) +
  scale_fill_brewer(palette = "Set3") +
  labs(title = "Resultados por Agrupación en cada Comuna",
       subtitle = "Elecciones CABA 2023 - Diputados/as",
       x = "Comuna",
       y = "Porcentaje de Votos",
       fill = "Partido Político") +
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),

  )

Mapeo de Resultados

Vamos a visualizar los resultados de Juntos por el Cambio en un mapa de CABA.

Code 
# Cargamo el GeoJSON de comunas de CABA que descargamos de Data CP
comunas_sf <- st_read("seccion_electorales_CABA.geojson", quiet = TRUE)

# Preparamos los datos usando resultados_partido y formateando el nombre de la comuna para que matchee con el GeoJSON
mapa_resultados <- resultados_partido %>%
  filter(agrupacion_nombre == "JUNTOS POR EL CAMBIO") %>%
  mutate(
    # Extraemos el número de comuna y lo formateamos
    comuna_num = as.integer(str_extract(seccion_nombre, "\\d+")),
    Seccion = paste("COMUNA", comuna_num)
  )

# Unimos los resultados con el GeoJSON
mapa_final <- comunas_sf %>%
  left_join(mapa_resultados, by = c("Seccion"))

# Creamos el mapa
ggplot() +
  geom_sf(data = mapa_final, aes(fill = porcentaje)) +
  scale_fill_gradient(
    low = "#FFF7BC",  # Amarillo claro
    high = "#D95F0E", # Naranja oscuro
    name = "% Votos JxC",
    na.value = "gray80"
  ) +
  geom_sf_text(
    data = mapa_final,
    aes(label = sprintf("%.1f%%", porcentaje)),
    size = 3
  ) +
  labs(
    title = "Resultados de Juntos por el Cambio",
    subtitle = "Porcentaje por Comuna - Legislativas CABA 2023"
  ) +
  theme_void() +  
  theme(
    plot.title = element_text(hjust = 0.5),
    plot.subtitle = element_text(hjust = 0.5),
    legend.position = "right"
  )

  kable(col.names = c("Tipo de Voto", "Total Votos", "Porcentaje"),
        align = c("l", "r", "r"))

# Calculamos las estadísticas generales
estadisticas_generales <- df_caba %>%
  summarise(
    total_mesas = n_distinct(mesa_id),
    total_electores = sum(mesa_electores[!duplicated(mesa_id)]),
    total_votos = sum(votos_cantidad),
    participacion = total_votos / total_electores * 100
  )

# Estadísticas generales
estadisticas_generales %>%
  mutate(
    total_mesas = format(total_mesas, big.mark = ".", decimal.mark = ","),
    total_electores = format(total_electores, big.mark = ".", decimal.mark = ","),
    total_votos = format(total_votos, big.mark = ".", decimal.mark = ","),
    participacion = sprintf("%.1f%%", participacion)
  ) %>%
  gather(key = "Estadística", value = "Valor") %>%
  mutate(
    Estadística = case_when(
      Estadística == "total_mesas" ~ "Total de Mesas",
      Estadística == "total_electores" ~ "Total de Electores",
      Estadística == "total_votos" ~ "Total de Votos Emitidos",
      Estadística == "participacion" ~ "Participación Total",
      TRUE ~ Estadística
    )
  ) %>%
  kable(align = c("l", "r"))
